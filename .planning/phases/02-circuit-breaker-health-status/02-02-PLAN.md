Integrate pybreaker circuit breakers into all provider searchers and create the HealthChecker.

**Purpose:** Circuit breaker wraps searcher public methods (search, fetch_metadata, download) so that when a provider server is unreachable, the breaker opens after failing `fail_max` times and subsequent calls fail fast. The HealthChecker provides lightweight health probes for the status command.

**Output:** HealthChecker class, breaker-wrapped NCBI and UniProt searchers, breaker_state property on DatabaseSearcher, breaker config wiring through Biocurator initialization.

    ```python
    """
    HealthChecker module for probing biological database availability.
    """

    import time
    import logging
    from dataclasses import dataclass
    from typing import Callable

    logger = logging.getLogger(__name__)


    @dataclass
    class HealthStatus:
        """Result of a health check probe."""
        provider: str
        reachable: bool
        response_time_ms: float = 0.0
        error: str | None = None


    class HealthChecker:
        """Probes provider API availability with lightweight queries.

        Per D-04: Uses minimal search queries guaranteed to return results.
        Per D-05: Response time measured in milliseconds, uses configured timeout.
        """

        @staticmethod
        def ping_ncbi(timeout: int = 30) -> HealthStatus:
            """Probe NCBI Entrez availability with minimal esearch.

            Uses Entrez.esearch(db="nuccore", term="a[organism]", retmax=1)
            which is a single-index search guaranteed to return results.
            """
            from Bio import Entrez

            start = time.monotonic()
            try:
                Entrez.email = "health-check@biocurator"
                handle = Entrez.esearch(
                    db="nuccore",
                    term="a[organism]",
                    retmax=1,
                )
                # Read to trigger the actual HTTP call
                from Bio.Entrez.Parser import DictionaryElement
                result: DictionaryElement = Entrez.read(handle)
                handle.close()
                elapsed = (time.monotonic() - start) * 1000
                _ = result.get("IdList", [])
                logger.debug(f"NCBI health check OK ({elapsed:.0f}ms)")
                return HealthStatus(
                    provider="ncbi",
                    reachable=True,
                    response_time_ms=round(elapsed, 1),
                )
            except Exception as exc:
                elapsed = (time.monotonic() - start) * 1000
                logger.warning(f"NCBI health check failed: {exc}")
                return HealthStatus(
                    provider="ncbi",
                    reachable=False,
                    response_time_ms=round(elapsed, 1),
                    error=str(exc),
                )

        @staticmethod
        def ping_uniprot(timeout: int = 30) -> HealthStatus:
            """Probe UniProt REST API availability with minimal search.

            Uses GET /uniprotkb/search?query=a&size=1 which is a
            lightweight index lookup guaranteed to return results.
            """
            import requests

            start = time.monotonic()
            try:
                response = requests.get(
                    "https://rest.uniprot.org/uniprotkb/search",
                    params={"query": "a", "size": 1},
                    timeout=timeout,
                )
                response.raise_for_status()
                elapsed = (time.monotonic() - start) * 1000
                logger.debug(f"UniProt health check OK ({elapsed:.0f}ms)")
                return HealthStatus(
                    provider="uniprot",
                    reachable=True,
                    response_time_ms=round(elapsed, 1),
                )
            except Exception as exc:
                elapsed = (time.monotonic() - start) * 1000
                logger.warning(f"UniProt health check failed: {exc}")
                return HealthStatus(
                    provider="uniprot",
                    reachable=False,
                    response_time_ms=round(elapsed, 1),
                    error=str(exc),
                )
    ```

    ```python
    def _init_breaker(self) -> pybreaker.CircuitBreaker | None:
        """Create and return a pybreaker CircuitBreaker from config.

        Per D-03 (CONTEXT.md): Only DatabaseSearchError and unhandled exceptions
        count as breaker failures. Non-retryable exceptions (ValueError, KeyError,
        4xx HTTP errors) are excluded via pybreaker's exclude list.

        The exclude list uses the same _is_retryable() predicate that tenacity uses
        — non-retryable exceptions never trip the breaker.
        """
        if self.config.breaker is None:
            return None

        resolved = self.config.breaker.resolve()
        return pybreaker.CircuitBreaker(
            fail_max=resolved.fail_max,
            reset_timeout=resolved.recovery_timeout,
            exclude=[
                ValueError,
                KeyError,
                TypeError,
            ],
        )

    @property
    def breaker_state(self) -> str | None:
        """Return the current circuit breaker state string.

        Returns one of: 'closed', 'open', 'half_open', or None if no breaker configured.
        """
        if not hasattr(self, '_breaker') or self._breaker is None:
            return None
        return self._breaker.state
    ```

    The `_init_breaker()` method does NOT call `pybreaker.CircuitBreakerListener` directly — per CONTEXT.md agent's discretion, logging state changes is expected. Add a logging listener as a nested class or inline. For simplicity, pybreaker's built-in state is sufficient; state transitions will be logged at the call site.
