    **Replace `_safe_get`:**
    Remove the `@retry(exceptions=(Exception,), max_attempts=3)` decorator. Rewrite as a regular method using `Retrying` class at call time:
    ```python
    def _safe_get(self, url: str, **kwargs):
        """Execute a GET request with retry logic.
        
        Only network-level exceptions (ConnectionError, Timeout, 5xx) are retried.
        4xx HTTP errors are NOT retried — they fail immediately.
        """
        retry_cfg = self.config.retry.resolve() if self.config.retry else RetryConfig.defaults()
        retryer = Retrying(
            stop=stop_after_attempt(retry_cfg.max_attempts),
            wait=wait_exponential(
                multiplier=retry_cfg.backoff_factor,
                max=retry_cfg.max_delay,
            ),
            retry=retry_if_exception_type(RETRYABLE_EXCEPTIONS),
            reraise=True,
            before_sleep=before_sleep_log(logger, logging.WARNING),
        )
        response = retryer(self.session.get, url, **kwargs)
        # raise_for_status is OUTSIDE the retry — 4xx raises here and is NOT retried
        response.raise_for_status()
        return response
    ```
    Key: `self.session.get(url, **kwargs)` is inside the retry. `response.raise_for_status()` is OUTSIDE — 4xx errors (client faults) will not be retried, while 5xx errors raised by `session.get` as `HTTPError` will be retried by tenacity through the `retry_if_exception_type(RETRYABLE_EXCEPTIONS)` predicate.

    **Rewrite `search()`:**
    Replace the `except Exception ... return []` with:
    ```python
    def search(self, criteria: UniProtSearchCriteria) -> list[str]:
        logger.info("Searching UniProt database...")
        query = self.build_query(criteria)
        try:
            url = f"{self._base_url}/uniprotkb/search"
            params = {
                "query": query,
                "format": "tsv",
                "fields": "accession",
                "size": min(criteria.max_results, 500),
            }
            response = self._safe_get(url, params=params, timeout=self.config.timeout)
            lines = response.text.strip().split("\n")[1:]
            ids = [line.strip() for line in lines if line.strip()]
            logger.info(f"Found {len(ids)} UniProt entries")
            return ids
        except DatabaseSearchError:
            raise
        except RETRYABLE_EXCEPTIONS as exc:
            raise DatabaseSearchError(f"UniProt search failed: {exc}") from exc
        except Exception as exc:
            raise DatabaseSearchError(f"UniProt search error: {exc}") from exc
    ```

    def test_search_raises_database_search_error_on_failure(searcher):
        """search() raises DatabaseSearchError instead of returning []."""
        from biocurator.exceptions import DatabaseSearchError
        from requests import ConnectionError
        with patch.object(searcher, "_safe_get", side_effect=ConnectionError("connection refused")):
            with pytest.raises(DatabaseSearchError, match="UniProt search failed"):
                searcher.search(UniProtSearchCriteria())


    def test_search_raises_on_bad_response(searcher):
        """4xx response (raise_for_status) in search raises DatabaseSearchError."""
        from biocurator.exceptions import DatabaseSearchError
        import requests
        # _safe_get retry only retries RETRYABLE_EXCEPTIONS, so a 4xx HTTPError
        # raised by raise_for_status() (outside retry) propagates as-is
        resp = type("FakeResponse", (), {"status_code": 400, "raise_for_status": lambda: (_ for _ in ()).throw(requests.HTTPError("400 Bad Request"))})()
        with patch.object(searcher, "_safe_get", side_effect=requests.HTTPError("400 Bad Request", response=resp)):
            with pytest.raises(DatabaseSearchError, match="UniProt search"):
                searcher.search(UniProtSearchCriteria())


    def test_fetch_metadata_continues_on_per_batch_error(searcher):
        """fetch_metadata logs warning and continues on per-batch failure."""
        from unittest.mock import patch
        with patch.object(searcher, "_safe_get", side_effect=ValueError("bad data")):
            results = list(searcher.fetch_metadata(["id1", "id2"], UniProtSearchCriteria()))
            assert results == []


    def test_download_continues_on_per_record_error(searcher):
        """download logs warning and continues on per-record failure."""
        from unittest.mock import patch

    import pytest
    from biocurator.config.schema import RetryConfig


    def test_retry_config_defaults():
        cfg = RetryConfig.defaults()
        assert cfg.max_attempts == 3
        assert cfg.backoff_factor == 2.0
        assert cfg.max_delay == 60
        assert cfg.timeout == 30


    def test_retry_config_resolve_none_fields():
        """resolve() fills None fields with defaults."""
        cfg = RetryConfig(max_attempts=5)
        resolved = cfg.resolve()
        assert resolved.max_attempts == 5  # overridden
        assert resolved.backoff_factor == 2.0  # default
        assert resolved.max_delay == 60  # default
        assert resolved.timeout == 30  # default


    def test_retry_config_resolve_with_custom_defaults():
        """resolve() uses provided defaults when field is None."""
        cfg = RetryConfig(max_attempts=7)
        base = RetryConfig(max_attempts=3, backoff_factor=3.0, max_delay=120, timeout=60)
        resolved = cfg.resolve(base)
        assert resolved.max_attempts == 7  # override
        assert resolved.backoff_factor == 3.0  # from base
        assert resolved.max_delay == 120  # from base
        assert resolved.timeout == 60  # from base


    def test_retry_config_from_dict_none():
        assert RetryConfig.from_dict(None) is None
        assert RetryConfig.from_dict({}) is None


    def test_retry_config_from_dict_partial():
        cfg = RetryConfig.from_dict({"max_attempts": 5})
        assert cfg is not None
        assert cfg.max_attempts == 5
        assert cfg.backoff_factor is None  # not set
        assert cfg.max_delay is None
        assert cfg.timeout is None


    def test_retry_config_from_dict_full():
        cfg = RetryConfig.from_dict({
            "max_attempts": 5,
            "backoff_factor": 3.0,
            "max_delay": 120,
            "timeout": 60,
        })
        assert cfg is not None
        assert cfg.max_attempts == 5
        assert cfg.backoff_factor == 3.0
        assert cfg.max_delay == 120
        assert cfg.timeout == 60


    def test_merge_override_wins():
        """Job-level override takes precedence over global default."""
        global_cfg = RetryConfig(max_attempts=3, backoff_factor=2.0)
        job_override = RetryConfig(max_attempts=5)
        merged = global_cfg.resolve(job_override)
        assert merged.max_attempts == 5  # override wins
        assert merged.backoff_factor == 2.0  # from global


    def test_no_override_uses_global():
        """Without job override, global defaults apply."""
        global_cfg = RetryConfig(max_attempts=3, backoff_factor=2.0)
        merged = global_cfg.resolve(None)
        assert merged.max_attempts == 3
        assert merged.backoff_factor == 2.0
